# 设备监控模块 - 前端组件详细设计

**版本**: 1.0  
**日期**: 2026-04-01  
**作者**: 工作流架构师

**响应式支持**: Desktop (≥1024px), Tablet (768px-1023px), Mobile (≤767px)

## 1. 设计概述

### 1.1 设计目标
设备监控前端模块提供直观、实时、高效的设备监控界面，支持：
- **实时监控**: 设备状态、运行数据的实时可视化
- **告警管理**: 告警的实时通知、确认、处理闭环
- **设备管理**: 设备的增删改查、配置、维护
- **数据分析**: 历史数据查询、趋势分析、报表
- **移动适配**: 支持移动端巡检和维护作业

### 1.2 设计原则
1. **实时性**: 关键数据秒级更新，操作反馈即时
2. **直观性**: 数据可视化，状态一目了然
3. **一致性**: 组件、交互、样式统一规范
4. **响应式**: 全设备兼容，移动端优先考虑
5. **性能**: 大数据量下的流畅体验
6. **可访问性**: 支持键盘导航、屏幕阅读器

### 1.3 技术约束
- **框架**: Vue 3.4 Composition API + `<script setup>`
- **语言**: TypeScript 5.3 (严格模式)
- **UI库**: Element Plus 2.7 + 自定义主题
- **状态管理**: Pinia (替代Vuex)
- **路由**: Vue Router 4
- **HTTP客户端**: Axios + 自定义封装
- **图表**: ECharts 5 (Apache 2.0)
- **地图**: 百度地图/高德地图API
- **WebSocket**: 原生WebSocket + 自动重连

#### 3.2.5 WebSocket集成
```typescript
class RealtimeDataService {
  private ws: WebSocket | null = null;
  private reconnectTimer: any = null;
  
  // 连接WebSocket
  connect(deviceId: string, onMessage: (data: any) => void) {
    const token = localStorage.getItem('access_token');
    const wsUrl = `${import.meta.env.VITE_WS_URL}/devices/${deviceId}/metrics?token=${token}`;
    
    this.ws = new WebSocket(wsUrl);
    
    this.ws.onopen = () => {
      console.log('WebSocket连接已建立');
      // 发送订阅消息
      this.ws?.send(JSON.stringify({
        action: 'subscribe',
        metrics: ['voltage', 'current', 'active_power'],
        interval: 1000
      }));
    };
    
    this.ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      onMessage(data);
    };
    
    this.ws.onclose = () => {
      console.log('WebSocket连接关闭，5秒后重连...');
      this.reconnectTimer = setTimeout(() => {
        this.connect(deviceId, onMessage);
      }, 5000);
    };
  }
  
  // 断开连接
  disconnect() {
    if (this.reconnectTimer) {
      clearTimeout(this.reconnectTimer);
    }
    if (this.ws) {
      this.ws.close();
      this.ws = null;
    }
  }
}
```

#### 3.3.4 声音提醒策略
```typescript
class NotificationSound {
  private audioContext: AudioContext | null = null;
  
  // 播放告警声音
  playAlertSound(level: AlertLevel) {
    if (!this.audioContext) {
      this.audioContext = new (window.AudioContext || (window as any).webkitAudioContext)();
    }
    
    const oscillator = this.audioContext.createOscillator();
    const gainNode = this.audioContext.createGain();
    
    // 根据告警级别设置频率
    let frequency = 800;
    switch(level) {
      case 'critical': frequency = 1200; break;
      case 'error': frequency = 1000; break;
      case 'warning': frequency = 800; break;
      default: return; // info级别不播放声音
    }
    
    oscillator.frequency.setValueAtTime(frequency, this.audioContext.currentTime);
    oscillator.type = 'sine';
    
    // 音量淡入淡出
    gainNode.gain.setValueAtTime(0, this.audioContext.currentTime);
    gainNode.gain.linearRampToValueAtTime(0.3, this.audioContext.currentTime + 0.1);
    gainNode.gain.linearRampToValueAtTime(0, this.audioContext.currentTime + 0.5);
    
    oscillator.connect(gainNode);
    gainNode.connect(this.audioContext.destination);
    
    oscillator.start();
    oscillator.stop(this.audioContext.currentTime + 0.5);
  }
}
```

## 4. 状态管理设计

### 4.1 Pinia Store设计

#### 4.1.1 设备Store (useDeviceStore)
```typescript
// stores/device.ts
export const useDeviceStore = defineStore('device', {
  state: () => ({
    // 设备列表
    devices: [] as Device[],
    devicesLoading: false,
    devicesError: null as string | null,
    
    // 当前选中设备
    currentDevice: null as Device | null,
    currentDeviceLoading: false,
    
    // 设备实时数据缓存
    realtimeData: new Map<string, RealtimeData>(),
    
    // 设备状态订阅
    subscriptions: new Map<string, WebSocket>(),
    
    // 筛选条件
    filters: {
      stationId: null as string | null,
      status: null as string | null,
      typeId: null as string | null,
      search: '',
    },
    
    // 分页信息
    pagination: {
      page: 1,
      pageSize: 20,
      total: 0,
    },
  }),
  
  getters: {
    // 筛选后的设备列表
    filteredDevices: (state) => {
      let devices = [...state.devices];
      
      // 应用筛选条件
      if (state.filters.stationId) {
        devices = devices.filter(d => d.stationId === state.filters.stationId);
      }
      if (state.filters.status) {
        devices = devices.filter(d => d.status === state.filters.status);
      }
      if (state.filters.typeId) {
        devices = devices.filter(d => d.deviceTypeId === state.filters.typeId);
      }
      if (state.filters.search) {
        const search = state.filters.search.toLowerCase();
        devices = devices.filter(d => 
          d.code.toLowerCase().includes(search) || 
          d.name.toLowerCase().includes(search)
        );
      }
      
      return devices;
    },
    
    // 分页后的设备列表
    pagedDevices: (state) => {
      const start = (state.pagination.page - 1) * state.pagination.pageSize;
      const end = start + state.pagination.pageSize;
      return state.filteredDevices.slice(start, end);
    },
    
    // 设备状态统计
    statusStats: (state) => {
      const stats = {
        online: 0,
        offline: 0,
        fault: 0,
        maintenance: 0,
        testing: 0,
        standby: 0,
      };
      
      state.devices.forEach(device => {
        if (stats.hasOwnProperty(device.status)) {
          stats[device.status as keyof typeof stats]++;
        }
      });
      
      return stats;
    },
  },
  
  actions: {
    // 加载设备列表
    async loadDevices(params?: LoadDevicesParams) {
      this.devicesLoading = true;
      this.devicesError = null;
      
      try {
        const response = await deviceApi.getDevices(params);
        this.devices = response.data;
        this.pagination = {
          page: response.meta.page,
          pageSize: response.meta.pageSize,
          total: response.meta.total,
        };
      } catch (error) {
        this.devicesError = error.message;
        throw error;
      } finally {
        this.devicesLoading = false;
      }
    },
    
    // 订阅设备实时数据
    subscribeToDevice(deviceId: string) {
      if (this.subscriptions.has(deviceId)) {
        return; // 已订阅
      }
      
      const ws = new RealtimeDataService();
      ws.connect(deviceId, (data) => {
        // 更新实时数据缓存
        this.realtimeData.set(deviceId, {
          ...data,
          updatedAt: new Date().toISOString(),
        });
      });
      
      this.subscriptions.set(deviceId, ws);
    },
    
    // 取消订阅
    unsubscribeFromDevice(deviceId: string) {
      const ws = this.subscriptions.get(deviceId);
      if (ws) {
        ws.disconnect();
        this.subscriptions.delete(deviceId);
        this.realtimeData.delete(deviceId);
      }
    },
  },
});
```

        alertId: alert.id,
        deviceId: alert.deviceId,
        deviceName: alert.deviceName,
        title: alert.title,
        message: alert.message,
        level: alert.level,
        timestamp: alert.startTime,
        read: false,
        acknowledged: false,
        actions: [
          {
            label: '确认',
            type: 'primary',
            handler: () => this.acknowledgeAlert(alert.id),
          },
          {
            label: '查看',
            type: 'default',
            handler: () => router.push(`/devices/${alert.deviceId}/alerts`),
          },
        ],
      };
      
      this.realtimeAlerts.unshift(notification);
      this.unreadCount++;
      
      // 限制最多保存50条实时告警
      if (this.realtimeAlerts.length > 50) {
        this.realtimeAlerts.pop();
      }
      
      // 更新统计
      this.updateStats();
    },
    
    // 确认告警
    async acknowledgeAlert(alertId: string) {
      try {
        await alertApi.acknowledgeAlert(alertId);
        
        // 更新本地状态
        const alert = this.alerts.find(a => a.id === alertId);
        if (alert) {
          alert.status = 'acknowledged';
          alert.acknowledgedAt = new Date().toISOString();
        }
        
        // 更新实时告警
        const notification = this.realtimeAlerts.find(n => n.alertId === alertId);
        if (notification) {
          notification.acknowledged = true;
        }
        
        this.updateStats();
      } catch (error) {
        console.error('确认告警失败:', error);
        throw error;
      }
    },
  },
});
