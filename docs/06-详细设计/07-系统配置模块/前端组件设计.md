# 系统配置模块 - 前端组件设计

## 概述
本文档描述系统配置模块的前端组件设计，包括系统参数管理、字典数据管理、审批流程配置、通知模板管理和日志配置等功能的界面组件、交互设计和状态管理。

## 技术栈
- **前端框架**: Vue 3 + TypeScript
- **UI组件库**: Element Plus
- **状态管理**: Pinia
- **路由管理**: Vue Router
- **HTTP客户端**: Axios
- **构建工具**: Vite

## 设计原则
1. **组件化**：遵循单一职责原则，每个组件只负责一个功能
2. **可复用性**：公共组件抽象为通用组件，支持多场景使用
3. **响应式设计**：支持PC端和移动端适配
4. **用户体验**：操作流程简洁明了，提供即时反馈
5. **可访问性**：支持键盘操作和屏幕阅读器

## 状态管理设计

### 1. 配置管理Store
```typescript
export const useConfigStore = defineStore('config', {
  state: () => ({
    configs: [] as ConfigItem[],
    configCache: new Map<string, ConfigItem>(),
    loading: false,
    searchParams: defaultSearchParams,
    pagination: defaultPagination
  }),
  
  actions: {
    // 异步获取配置列表
    async fetchConfigs(params?: SearchParams) {
      this.loading = true;
      try {
        const response = await configApi.getConfigs(params);
        this.configs = response.data.list;
        this.pagination = {
          page: response.data.page,
          size: response.data.size,
          total: response.data.total
        };
      } finally {
        this.loading = false;
      }
    },
    
    // 获取单个配置（优先从缓存读取）
    async getConfig(configKey: string, forceRefresh = false) {
      const cacheKey = `${this.tenantId}:${configKey}`;
      
      // 检查缓存
      if (!forceRefresh && this.configCache.has(cacheKey)) {
        return this.configCache.get(cacheKey);
      }
      
      // 从API获取
      const config = await configApi.getConfig(configKey);
      this.configCache.set(cacheKey, config);
      return config;
    },
    
    // 更新配置
    async updateConfig(config: ConfigItem) {
      const updated = await configApi.updateConfig(config);
      
      // 更新本地状态
      const index = this.configs.findIndex(c => c.configKey === config.configKey);
      if (index >= 0) {
        this.configs[index] = updated;
      }
      
      // 更新缓存
      const cacheKey = `${this.tenantId}:${config.configKey}`;
      this.configCache.set(cacheKey, updated);
      
      // 发送配置更新事件
      emitter.emit('config:updated', updated);
      
      return updated;
    }
  }
});
```

### 2. 字典数据Store
```typescript
export const useDictStore = defineStore('dict', {
  state: () => ({
    // 按字典类型组织的字典数据
    dictData: new Map<string, DictItem[]>(),
    // 字典类型列表
    dictTypes: [] as string[],
    // 加载状态
    loading: false
  }),
  
  actions: {
    // 获取字典类型
    async fetchDictTypes() {
      this.loading = true;
      try {
        this.dictTypes = await dictApi.getDictTypes();
      } finally {
        this.loading = false;
      }
    },
    
    // 获取字典数据
    async fetchDictData(dictType: string, forceRefresh = false) {
      // 检查是否已有缓存
      if (!forceRefresh && this.dictData.has(dictType)) {
        return this.dictData.get(dictType);
      }
      
      this.loading = true;
      try {
        const data = await dictApi.getDictData(dictType);
        this.dictData.set(dictType, data);
        return data;
      } finally {
        this.loading = false;
      }
    },
    
    // 获取字典项（支持树形结构）
    getDictTree(dictType: string): DictItem[] {
      const data = this.dictData.get(dictType) || [];
      return this.buildTree(data);
    },
    
    // 构建树形结构
    private buildTree(items: DictItem[]): DictItem[] {
      const tree: DictItem[] = [];
      const map = new Map<number, DictItem>();
      
      // 创建映射
      items.forEach(item => {
        map.set(item.id, { ...item, children: [] });
      });
      
      // 构建树
      items.forEach(item => {
        const node = map.get(item.id)!;
        if (item.parentId && map.has(item.parentId)) {
          map.get(item.parentId)!.children!.push(node);
        } else {
          tree.push(node);
        }
      });
      
      return tree;
    }
  }
});
```
