## 1. 安全设计概述

### 1.1 安全目标
工单管理模块作为NESOM系统的业务核心，处理设备维护全流程数据，必须满足以下安全目标：

1. **数据机密性**: 防止未授权访问工单数据、设备信息、人员信息
2. **操作完整性**: 确保工单状态流转、备件消耗、成本核算数据不被篡改
3. **业务可用性**: 保障工单处理流程7×24小时可靠运行
4. **操作可追溯**: 所有工单操作可审计、可追溯、不可抵赖
5. **合规性**: 符合等保2.0三级、个人信息保护法、数据安全法要求

### 1.2 安全威胁模型

#### 1.2.1 威胁识别
| 威胁类型 | 威胁描述 | 影响等级 | 可能性 |
|----------|----------|----------|--------|
| 工单数据泄露 | 攻击者获取工单详情、设备信息、人员信息 | 严重 | 中 |
| 工单操作篡改 | 恶意修改工单状态、处理结果、成本数据 | 严重 | 中 |
| 权限滥用 | 内部人员越权操作工单，如分配给自己 | 高 | 中 |
| 拒绝服务 | 攻击导致工单系统不可用，影响设备维护 | 高 | 中 |
| 移动端数据泄露 | 移动端设备丢失或被盗导致数据泄露 | 中 | 高 |
| 扫码攻击 | 伪造工单二维码进行恶意操作 | 中 | 中 |
| 供应链攻击 | 通过第三方组件入侵工单系统 | 中 | 低 |

#### 1.2.2 攻击面分析
1. **Web API**: RESTful API端点、WebSocket实时通知
2. **移动端应用**: 微信小程序、H5、App客户端
3. **数据库**: 工单数据存储和查询
4. **文件存储**: 工单附件、现场照片存储
5. **二维码生成和解析**: 工单扫码功能
6. **位置服务**: 移动端定位和地理围栏
7. **消息推送**: 工单状态变更通知

### 1.3 安全设计原则
1. **最小权限原则**: 用户仅拥有完成工作所需的最小权限
2. **职责分离**: 工单创建、分配、处理、审核由不同角色负责
3. **数据脱敏**: 敏感数据在显示和传输时脱敏处理
4. **操作审计**: 所有工单操作记录详细审计日志
5. **输入验证**: 所有用户输入进行严格验证和过滤
6. **安全默认配置**: 默认配置即为安全配置，减少误配置风险

```python
class WorkOrderAccessPolicy:
    def can_view_work_order(self, user, work_order):
        """判断用户是否可以查看工单"""
        
        # 规则1: 管理员可以查看所有工单
        if user.has_role('admin'):
            return True
            
        # 规则2: 工单创建者可以查看自己的工单
        if work_order.reported_by == user.id:
            return True
            
        # 规则3: 分配的执行人可以查看分配给自己的工单
        if work_order.assigned_to == user.id:
            return True
            
        # 规则4: 同一场站的管理员可以查看该场站的工单
        if user.has_role('station_admin') and work_order.station_id in user.station_scope:
            return True
            
        # 规则5: 相同部门的主管可以查看部门内工单
        if user.has_role('department_manager') and work_order.department_id == user.department_id:
            return True
            
        return False
    
    def can_update_work_order(self, user, work_order):
        """判断用户是否可以更新工单"""
        
        # 状态约束: 已关闭工单不可更新
        if work_order.status == 'closed':
            return False
            
        # 时间约束: 超过修改时限不可更新
        if work_order.status != 'draft' and (datetime.now() - work_order.created_at).hours > 24:
            return False
            
        # 权限检查
        return self.can_view_work_order(user, work_order) and user.has_permission('work_order:update')
```

#### 2.2.3 数据级权限控制
1. **场站隔离**: 用户只能访问有权限的场站工单
2. **部门隔离**: 部门主管只能查看本部门工单
3. **时间隔离**: 只能修改最近24小时内创建的工单
4. **状态隔离**: 已关闭工单只读，草稿工单只有创建者可修改

```typescript
// 数据脱敏处理器
class DataMaskingHandler {
  static maskWorkOrder(workOrder: WorkOrder, user: User): WorkOrderDTO {
    const dto = { ...workOrder };
    
    // 脱敏规则
    if (!user.hasPermission('work_order:view_sensitive')) {
      // 隐藏成本信息
      delete dto.costEstimate;
      delete dto.actualCost;
      
      // 隐藏联系人详细信息
      if (dto.reportedBy) {
        dto.reportedBy.phone = this.maskPhone(dto.reportedBy.phone);
        dto.reportedBy.email = this.maskEmail(dto.reportedBy.email);
      }
    }
    
    // 场站隔离
    if (!user.stationScope.includes(workOrder.stationId)) {
      throw new ForbiddenError('无权访问该场站工单');
    }
    
    return dto;
  }
  
  static maskPhone(phone: string): string {
    return phone.replace(/(\d{3})\d{4}(\d{4})/, '$1****$2');
  }
  
  static maskEmail(email: string): string {
    const [name, domain] = email.split('@');
    return `${name.charAt(0)}****${name.charAt(name.length-1)}@${domain}`;
  }
}
```

```python
class SecurityHeadersMiddleware:
    def __init__(self, app):
        self.app = app
    
    async def __call__(self, scope, receive, send):
        async def send_with_headers(message):
            if message['type'] == 'http.response.start':
                # 添加安全头
                headers = message.get('headers', [])
                headers.extend([
                    (b'X-Content-Type-Options', b'nosniff'),

            await send(message)
        
        await self.app(scope, receive, send_with_headers)
```

#### 5.1.1 状态转换验证
```python
class WorkOrderStateMachine:
    VALID_TRANSITIONS = {
        'draft': ['pending', 'cancelled'],
        'pending': ['assigned', 'cancelled'],
        'assigned': ['in_progress', 'cancelled'],
        'in_progress': ['pending_review', 'cancelled'],
        'pending_review': ['completed', 'in_progress', 'cancelled'],
        'completed': ['closed', 'cancelled'],
        'closed': [],  # 终态
        'cancelled': [],  # 终态
    }
    
    def can_transition(self, from_state, to_state, user, work_order):
        # 1. 验证状态转换是否允许
        if to_state not in self.VALID_TRANSITIONS.get(from_state, []):
            return False, f"不允许从{from_state}转换到{to_state}"
        
        # 2. 验证用户权限
        if not self.has_transition_permission(from_state, to_state, user):
            return False, "用户无权限进行此状态转换"
        
        # 3. 验证业务规则
        if not self.validate_business_rules(from_state, to_state, work_order):
            return False, "不符合业务规则"
        
        # 4. 验证数据完整性
        if not self.validate_data_integrity(from_state, to_state, work_order):
            return False, "数据不完整"
        
        return True, None
    
    def validate_business_rules(self, from_state, to_state, work_order):
        # 特定状态转换的业务规则
        
        if from_state == 'in_progress' and to_state == 'pending_review':
            # 提交审核前必须有处理记录
            if not work_order.details or len(work_order.details) == 0:
                return False
            
            # 必须填写完成时间和完成度
            if not work_order.actual_end or work_order.completion_rate < 100:
                return False
        
        elif from_state == 'pending_review' and to_state == 'completed':
            # 审核通过前必须有审核意见
            if not work_order.review_notes:
                return False
            
            # 必须有实际成本记录
            if not work_order.actual_cost:
                return False
        
        return True
```

#### 5.1.2 防重复提交
```typescript
class IdempotencyHandler {
  private store: Redis;  // Redis存储
  
  async ensureIdempotency(
    requestId: string, 
    userId: string, 
    action: string,
    handler: () => Promise<any>
  ): Promise<any> {
    const key = `idempotency:${userId}:${action}:${requestId}`;
    
    // 检查是否已处理
    const existing = await this.store.get(key);
    if (existing) {
      // 返回之前的结果
      return JSON.parse(existing);
    }
    
    // 设置处理中标记（防并发）
    const lockAcquired = await this.store.setnx(`${key}:lock`, '1');
    if (!lockAcquired) {
      throw new ConflictError('请求正在处理中');
    }
    
    try {
      // 执行处理
      const result = await handler();
      
      // 存储结果（24小时过期）
      await this.store.setex(key, 24 * 60 * 60, JSON.stringify(result));
      
      return result;
    } finally {
      // 释放锁
      await this.store.del(`${key}:lock`);
    }
  }
}
```

```typescript
  async setSecureItem(key: string, data: any): Promise<void> {
    const encrypted = await this.encrypt(JSON.stringify(data));
    await localStorage.setItem(key, encrypted);
  }
  
  // 读取解密数据
  async getSecureItem(key: string): Promise<any> {
    const encrypted = await localStorage.getItem(key);
    if (!encrypted) return null;
    
    const decrypted = await this.decrypt(encrypted);
    return JSON.parse(decrypted);
  }
  
  // 加密方法
  private async encrypt(text: string): Promise<string> {
    const iv = crypto.getRandomValues(new Uint8Array(12));
    const algorithm = { name: 'AES-GCM', iv };
    
    const key = await crypto.subtle.importKey(
      'raw',
      this.encryptionKey,
      algorithm,
      false,
      ['encrypt']
    );
    
    const encrypted = await crypto.subtle.encrypt(
      algorithm,
      key,
      new TextEncoder().encode(text)
    );
    
    return btoa(String.fromCharCode(...iv, ...new Uint8Array(encrypted)));
  }
}
```

### 7.1 二维码安全设计
```typescript
class QRCodeSecurity {
  // 生成安全的工单二维码
  generateWorkOrderQR(workOrderId: string): string {
    const payload = {
      id: workOrderId,
      type: 'work_order',
      timestamp: Date.now(),
      expiresAt: Date.now() + 24 * 60 * 60 * 1000,  // 24小时有效
    };
    
    // 添加数字签名
    const signature = this.signPayload(payload);
    payload.signature = signature;
    
    // 转换为二维码内容
    return `nesom://workorder/${btoa(JSON.stringify(payload))}`;
  }
  
  // 验证二维码
  verifyQRCode(qrContent: string): { valid: boolean; payload?: any } {
    try {
      // 解析二维码
      const match = qrContent.match(/nesom:\/\/workorder\/(.+)/);
      if (!match) return { valid: false };
      
      const payloadStr = atob(match[1]);
      const payload = JSON.parse(payloadStr);
      
      // 检查过期时间
      if (payload.expiresAt < Date.now()) {
        return { valid: false, reason: '二维码已过期' };
      }
      
      // 验证签名
      const signature = payload.signature;
      delete payload.signature;
      
      if (!this.verifySignature(payload, signature)) {
        return { valid: false, reason: '签名无效' };
      }
      
      return { valid: true, payload };
      
    } catch (error) {
      return { valid: false, reason: '二维码格式错误' };
    }
  }
  
  private signPayload(payload: any): string {
    const payloadStr = JSON.stringify(payload);
    return crypto.createHmac('sha256', SECRET_KEY)
      .update(payloadStr)
      .digest('hex');
  }
}
```

```python
class AnomalyDetection:
    def detect_anomalies(self, user_id, actions):
        """检测用户异常行为"""
        
        anomalies = []
        
        # 1. 频率异常：短时间内大量操作
        recent_actions = [a for a in actions if a.timestamp > datetime.now() - timedelta(minutes=5)]
        if len(recent_actions) > 20:  # 5分钟内超过20次操作
            anomalies.append({
                'type': 'high_frequency',
                'count': len(recent_actions),
                'threshold': 20
            })
        
        # 2. 时间异常：非工作时间操作
        hour = datetime.now().hour
        if hour < 6 or hour > 22:  # 凌晨或深夜
            anomalies.append({
                'type': 'unusual_time',
                'hour': hour
            })
        
        # 3. 权限异常：尝试越权操作
        failed_auth_attempts = [a for a in actions if not a.success and 'permission' in a.error_message]
        if len(failed_auth_attempts) > 5:
            anomalies.append({
                'type': 'permission_abuse',
                'count': len(failed_auth_attempts)
            })
        
        # 4. 数据异常：异常数据修改模式
        cost_changes = [a for a in actions if a.event_type == 'cost_change']
        if cost_changes:
            total_increase = sum(max(0, (c.new_values.get('actual_cost', 0) or 0) - 
                                     (c.old_values.get('actual_cost', 0) or 0)) 
                                for c in cost_changes)
            if total_increase > 10000:  # 累计增加超过1万元
                anomalies.append({
                    'type': 'unusual_cost_increase',
                    'amount': total_increase
                })
        
        return anomalies
```
