## 1. 设计概述

### 1.1 设计目标
构建NESOM系统集成层的全方位安全防护体系，确保跨模块交互的安全性、可靠性和合规性，包括：
- 保障模块间通信的机密性、完整性和可用性
- 实现细粒度的跨模块访问控制和权限管理
- 建立完整的集成安全审计和监控机制
- 防范集成层特有的安全威胁和攻击向量
- 满足行业安全标准和法规合规要求

### 1.2 设计原则
1. **零信任原则**：不信任任何内部或外部实体，始终验证
2. **最小权限原则**：只授予完成工作所需的最小权限
3. **纵深防御原则**：多层安全防护，单一防线失效不影响整体安全
4. **默认安全原则**：安全配置默认开启，需要时显式关闭
5. **完整审计原则**：所有安全相关操作可追溯、可审计
6. **持续监控原则**：实时监控安全状态，及时发现和响应威胁

### 1.3 威胁模型分析
| 威胁类别 | 具体威胁 | 影响程度 | 发生概率 | 防护措施 |
|----------|----------|----------|----------|----------|
| 通信安全 | 中间人攻击、数据窃听 | 高 | 中 | TLS加密、证书固定 |
| 认证授权 | 凭证泄露、权限提升 | 高 | 中 | 多因素认证、动态令牌 |
| 数据安全 | 数据泄露、篡改 | 高 | 中 | 加密存储、完整性校验 |
| 拒绝服务 | 资源耗尽、服务瘫痪 | 中 | 低 | 限流熔断、资源隔离 |
| 注入攻击 | 代码注入、命令注入 | 高 | 中 | 输入验证、参数化查询 |
| 配置错误 | 默认凭证、开放端口 | 中 | 高 | 安全基线、自动检查 |

```python
# 服务间mTLS认证实现
class MutualTLSValidator:
    def __init__(self, ca_cert_path: str):
        self.ca_cert = self.load_certificate(ca_cert_path)
        self.cert_cache = {}  # 证书缓存
        
    def validate_client_certificate(self, client_cert_pem: str, 
                                   expected_cn: str = None) -> bool:
        """验证客户端证书"""
        try:
            # 1. 解析证书
            cert = x509.load_pem_x509_certificate(
                client_cert_pem.encode(), default_backend()
            )
            
            # 2. 验证证书链
            self.verify_certificate_chain(cert)
            
            # 3. 验证证书有效期
            if not self.is_certificate_valid(cert):
                raise CertificateExpiredError("证书已过期")
            
            # 4. 验证主题（可选）
            if expected_cn:
                cn = self.extract_common_name(cert)
                if cn != expected_cn:
                    raise CertificateValidationError(
                        f"证书CN不匹配: 期望{expected_cn}, 实际{cn}"
                    )
            
            # 5. 检查证书吊销状态
            if self.is_certificate_revoked(cert):
                raise CertificateRevokedError("证书已被吊销")
            
            # 6. 缓存验证结果
            cert_fingerprint = self.get_certificate_fingerprint(cert)
            self.cert_cache[cert_fingerprint] = {
                'cert': cert,
                'validated_at': datetime.now(),
                'valid_until': cert.not_valid_after
            }
            
            return True
            
        except Exception as e:
            logger.error(f"客户端证书验证失败: {e}")
            return False
    
    def get_client_identity(self, client_cert_pem: str) -> str:
        """从证书提取客户端身份"""
        cert = x509.load_pem_x509_certificate(
            client_cert_pem.encode(), default_backend()
        )
        
        # 提取SAN或CN作为身份标识
        san_ext = cert.extensions.get_extension_for_class(
            x509.SubjectAlternativeName
        )
        if san_ext:
            # 优先使用DNS名称
            dns_names = san_ext.value.get_values_for_type(x509.DNSName)
            if dns_names:
                return dns_names[0]
        
        # 回退到CN
        cn = self.extract_common_name(cert)
        return cn
```

### 2.2 消息层安全

#### 2.2.1 消息加密和签名
```python
# 安全消息封装
class SecureMessage:
    def __init__(self, encryption_key: bytes, signing_key: bytes):
        self.encryption_key = encryption_key
        self.signing_key = signing_key
        
    def encrypt_and_sign(self, message: dict) -> dict:
        """加密并签名消息"""
        # 1. 序列化消息
        message_json = json.dumps(message, separators=(',', ':'))
        message_bytes = message_json.encode('utf-8')
        
        # 2. 生成消息ID和时间戳
        message_id = str(uuid.uuid4())
        timestamp = int(time.time())
        
        # 3. 加密消息体（AES-GCM）
        nonce = os.urandom(12)
        cipher = AES.new(self.encryption_key, AES.MODE_GCM, nonce=nonce)
        ciphertext, tag = cipher.encrypt_and_digest(message_bytes)
        
        # 4. 签名消息头
        header = {
            'message_id': message_id,
            'timestamp': timestamp,
            'nonce': b64encode(nonce).decode('ascii'),
            'tag': b64encode(tag).decode('ascii')
        }
        
        header_json = json.dumps(header, separators=(',', ':'))
        signature = hmac.new(
            self.signing_key, 
            header_json.encode('utf-8'), 
            'sha256'
        ).digest()
        
        # 5. 返回安全消息
        return {
            'header': header,
            'signature': b64encode(signature).decode('ascii'),
            'ciphertext': b64encode(ciphertext).decode('ascii')
        }
    
    def verify_and_decrypt(self, secure_message: dict) -> dict:
        """验证签名并解密消息"""
        # 1. 验证签名
        header = secure_message['header']
        signature = b64decode(secure_message['signature'])
        
        header_json = json.dumps(header, separators=(',', ':'))
        expected_signature = hmac.new(
            self.signing_key,
            header_json.encode('utf-8'),
            'sha256'
        ).digest()
        
        if not hmac.compare_digest(signature, expected_signature):
            raise SignatureVerificationError("消息签名无效")
        
        # 2. 验证时间戳（防止重放攻击）
        timestamp = header['timestamp']
        current_time = int(time.time())
        if abs(current_time - timestamp) > 300:  # 5分钟窗口
            raise TimestampValidationError("消息时间戳超出允许范围")
        
        # 3. 解密消息体
        nonce = b64decode(header['nonce'])
        tag = b64decode(header['tag'])
        ciphertext = b64decode(secure_message['ciphertext'])
        
        cipher = AES.new(self.encryption_key, AES.MODE_GCM, nonce=nonce)
        message_bytes = cipher.decrypt_and_verify(ciphertext, tag)
        
        # 4. 反序列化消息
        message = json.loads(message_bytes.decode('utf-8'))
        
        # 5. 检查消息ID是否已使用（防止重放）
        message_id = header['message_id']
        if self.is_message_id_used(message_id):
            raise ReplayAttackError("消息ID已使用，疑似重放攻击")
        
        self.record_message_id(message_id)
        
        return message
```

```python
# 集成专用JWT令牌
class IntegrationJWT:
    def __init__(self, secret_key: str, issuer: str):
        self.secret_key = secret_key
        self.issuer = issuer
        
    def generate_token(self, module_name: str, permissions: list, 
                      ttl_minutes: int = 60) -> str:
        """生成集成JWT令牌"""
        now = datetime.utcnow()
        
        payload = {
            # 标准声明
            'iss': self.issuer,  # 签发者
            'sub': module_name,  # 主题（模块名称）
            'aud': 'integration',  # 受众
            'iat': int(now.timestamp()),  # 签发时间
            'exp': int((now + timedelta(minutes=ttl_minutes)).timestamp()),  # 过期时间
            'jti': str(uuid.uuid4()),  # 令牌ID
            
            # 自定义声明
            'module': module_name,
            'permissions': permissions,
            'token_type': 'integration',
            'scope': 'module_integration'
        }
        
        token = jwt.encode(payload, self.secret_key, algorithm='HS256')
        return token
    
    def validate_token(self, token: str) -> dict:
        """验证JWT令牌"""
        try:
            payload = jwt.decode(
                token,
                self.secret_key,
                algorithms=['HS256'],
                issuer=self.issuer,
                audience='integration'
            )
            
            # 验证自定义声明
            if payload.get('token_type') != 'integration':
                raise InvalidTokenError("令牌类型无效")
            
            if 'module' not in payload or 'permissions' not in payload:
                raise InvalidTokenError("令牌缺少必要声明")
            
            return payload
            
        except jwt.ExpiredSignatureError:
            raise TokenExpiredError("令牌已过期")
        except jwt.InvalidTokenError as e:
            raise InvalidTokenError(f"令牌无效: {str(e)}")
    
    def refresh_token(self, old_token: str) -> str:
        """刷新令牌（延长有效期）"""
        payload = self.validate_token(old_token)
        
        # 检查是否允许刷新
        if not payload.get('refreshable', False):
            raise TokenNotRefreshableError("此令牌不支持刷新")
        
        # 生成新令牌（保持相同权限）
        new_token = self.generate_token(
            module_name=payload['module'],
            permissions=payload['permissions'],
            ttl_minutes=60
        )
        
        # 记录令牌刷新
        self.record_token_refresh(
            old_jti=payload['jti'],
            new_jti=self.extract_jti(new_token)
        )
        
        return new_token
```

```python
# ABAC策略引擎
class IntegrationABAC:
    def __init__(self, policy_store):
        self.policy_store = policy_store
        self.decision_cache = {}  # 决策缓存
        
    def check_permission(self, request: AccessRequest) -> AccessDecision:
        """检查访问权限"""
        cache_key = self._generate_cache_key(request)
        
        # 检查缓存
        if cache_key in self.decision_cache:
            cached_decision = self.decision_cache[cache_key]
            if not cached_decision.is_expired():
                return cached_decision
        
        # 评估策略
        decision = self._evaluate_policies(request)
        
        # 缓存决策
        self.decision_cache[cache_key] = decision
        
        # 记录审计日志
        self._audit_access(request, decision)
        
        return decision
    
    def _evaluate_policies(self, request: AccessRequest) -> AccessDecision:
        """评估所有相关策略"""
        policies = self.policy_store.get_relevant_policies(request)
        
        if not policies:
            # 默认拒绝
            return AccessDecision(
                allowed=False,
                reason="无适用策略，默认拒绝",
                obligations=[]
            )
        
        # 按优先级排序
        policies.sort(key=lambda p: p.priority, reverse=True)
        
        # 应用策略
        for policy in policies:
            if self._policy_applies(policy, request):
                if policy.effect == "allow":
                    return AccessDecision(
                        allowed=True,
                        reason=f"策略允许: {policy.id}",
                        obligations=policy.obligations
                    )
                else:
                    return AccessDecision(
                        allowed=False,
                        reason=f"策略拒绝: {policy.id}",
                        obligations=[]
                    )
        
        # 默认拒绝
        return AccessDecision(
            allowed=False,
            reason="所有适用策略评估后拒绝",
            obligations=[]
        )
    
    def _policy_applies(self, policy: Policy, request: AccessRequest) -> bool:
        """检查策略是否适用"""
        # 检查主题匹配
        if not self._match_attribute(policy.subject, request.subject):
            return False
        
        # 检查资源匹配
        if not self._match_attribute(policy.resource, request.resource):
            return False
        
        # 检查操作匹配
        if not self._match_attribute(policy.action, request.action):
            return False
        
        # 检查环境条件
        if policy.conditions:
            for condition in policy.conditions:
                if not self._evaluate_condition(condition, request.context):
                    return False
        
        return True
```

## 4. 数据安全设计

### 4.1 数据分类和分级

#### 4.1.1 数据分类标准
| 数据类别 | 级别 | 描述 | 保护要求 |
|----------|------|------|----------|
| 用户凭证 | P0 | 密码、API密钥、令牌 | 加密存储、传输加密、访问日志 |
| 个人身份信息 | P1 | 姓名、电话、邮箱 | 加密存储、访问控制、脱敏显示 |
| 设备敏感数据 | P2 | 设备配置、密钥、证书 | 加密存储、传输加密、最小权限 |
| 业务运营数据 | P3 | 运行指标、告警数据 | 访问控制、完整性校验 |
| 系统监控数据 | P4 | 日志、性能指标 | 访问控制、防篡改 |

#### 4.1.2 数据生命周期安全
```python
# 数据生命周期安全管理
class DataLifecycleManager:
    def __init__(self, classification_rules):
        self.classification_rules = classification_rules
        
    def classify_data(self, data: dict, context: dict) -> DataClassification:
        """自动数据分类"""
        classification = DataClassification(level=4)  # 默认最低级别
        
        # 应用分类规则
        for rule in self.classification_rules:
            if self._rule_matches(rule, data, context):
                if rule.level < classification.level:  # 数字越小级别越高
                    classification = DataClassification(
                        level=rule.level,
                        category=rule.category,
                        protection_requirements=rule.protection_requirements
                    )
        
        return classification
    
    def apply_protection(self, data: dict, classification: DataClassification) -> ProtectedData:
        """应用数据保护措施"""
        protected_data = data.copy()
        
        # 根据分类级别应用保护
        if classification.level <= 1:  # P0-P1
            # 高强度加密
            protected_data = self._encrypt_sensitive_fields(
                protected_data, 
                classification.protection_requirements
            )
            
            # 添加完整性保护
            protected_data['_integrity_hash'] = self._calculate_integrity_hash(
                protected_data
            )
        
        elif classification.level <= 2:  # P2
            # 中等强度加密
            protected_data = self._encrypt_sensitive_fields(
                protected_data,
                classification.protection_requirements
            )
        
        # 添加元数据
        protected_data['_metadata'] = {
            'classified_at': datetime.now().isoformat(),
            'classification': classification.to_dict(),
            'protection_applied': True
        }
        
        return protected_data
    
    def sanitize_for_display(self, data: dict, user_role: str) -> dict:
        """数据脱敏显示"""
        sanitized = data.copy()
        
        # 根据用户角色应用脱敏规则
        if user_role == 'viewer':
            # 观众角色：脱敏所有敏感字段
            sanitized = self._mask_sensitive_fields(sanitized)
        elif user_role == 'operator':
            # 操作员角色：部分脱敏
            sanitized = self._partial_mask_sensitive_fields(sanitized)
        elif user_role == 'admin':
            # 管理员角色：不脱敏
            pass
        
        return sanitized
```

#### 4.2.1 字段级加密

```python
# 字段级加密实现
class FieldLevelEncryption:
    def __init__(self, kek_provider, dek_cache_size=1000):
        self.kek_provider = kek_provider  # 密钥加密密钥提供者
        self.dek_cache = LRUCache(maxsize=dek_cache_size)
        
    def encrypt_field(self, field_name: str, field_value: any, 
                     context: dict) -> EncryptedField:
        """加密字段"""
        # 1. 获取或生成数据加密密钥（DEK）
        dek_id, dek = self._get_or_create_dek(field_name, context)
        
        # 2. 序列化字段值
        serialized_value = self._serialize_value(field_value)
        
        # 3. 使用DEK加密数据
        cipher = AES.new(dek, AES.MODE_GCM)
        ciphertext, tag = cipher.encrypt_and_digest(serialized_value)
        
        # 4. 使用KEK加密DEK
        kek = self.kek_provider.get_kek(context)
        encrypted_dek = self._encrypt_dek(dek, kek)
        
        # 5. 返回加密字段结构
        return EncryptedField(
            encrypted_value=b64encode(ciphertext).decode('ascii'),
            encryption_metadata={
                'dek_id': dek_id,
                'encrypted_dek': b64encode(encrypted_dek).decode('ascii'),
                'nonce': b64encode(cipher.nonce).decode('ascii'),
                'tag': b64encode(tag).decode('ascii'),
                'algorithm': 'AES-256-GCM',
                'key_version': kek.version,
                'field_name': field_name,
                'encrypted_at': datetime.now().isoformat()
            }
        )
    
    def decrypt_field(self, encrypted_field: EncryptedField, 
                     context: dict) -> any:
        """解密字段"""
        # 1. 获取KEK
        kek = self.kek_provider.get_kek(context)
        
        # 2. 解密DEK
        encrypted_dek = b64decode(encrypted_field.encryption_metadata['encrypted_dek'])
        dek = self._decrypt_dek(encrypted_dek, kek)
        
        # 3. 解密数据
        ciphertext = b64decode(encrypted_field.encrypted_value)
        nonce = b64decode(encrypted_field.encryption_metadata['nonce'])
        tag = b64decode(encrypted_field.encryption_metadata['tag'])
        
        cipher = AES.new(dek, AES.MODE_GCM, nonce=nonce)
        serialized_value = cipher.decrypt_and_verify(ciphertext, tag)
        
        # 4. 反序列化字段值
        field_value = self._deserialize_value(serialized_value)
        
        return field_value
    
    def rekey_field(self, encrypted_field: EncryptedField, 
                   old_context: dict, new_context: dict) -> EncryptedField:
        """密钥轮换"""
        # 1. 解密字段
        field_value = self.decrypt_field(encrypted_field, old_context)
        
        # 2. 使用新上下文重新加密
        new_encrypted_field = self.encrypt_field(
            encrypted_field.encryption_metadata['field_name'],
            field_value,
            new_context
        )
        
        return new_encrypted_field
```

```python
# 实时威胁检测引擎
class ThreatDetectionEngine:
    def __init__(self, detection_rules, anomaly_models):
        self.detection_rules = detection_rules
        self.anomaly_models = anomaly_models
        self.event_window = deque(maxlen=1000)  # 事件窗口
        
    def analyze_event(self, event: dict) -> ThreatAssessment:
        """分析安全事件"""
        threat_level = 'normal'
        detection_results = []
        
        # 1. 基于规则的检测
        rule_detections = self._apply_detection_rules(event)
        if rule_detections:
            threat_level = self._calculate_threat_level(rule_detections)
            detection_results.extend(rule_detections)
        
        # 2. 基于异常的检测
        self.event_window.append(event)
        if len(self.event_window) >= 100:  # 有足够历史数据
            anomaly_detections = self._detect_anomalies(event)
            if anomaly_detections:
                threat_level = max(threat_level, 'suspicious')
                detection_results.extend(anomaly_detections)
        
        # 3. 上下文关联分析
        correlated_threats = self._correlate_threats(event)
        if correlated_threats:
            threat_level = max(threat_level, 'high')
            detection_results.extend(correlated_threats)
        
        # 4. 生成威胁评估
        assessment = ThreatAssessment(
            threat_level=threat_level,
            detections=detection_results,
            event=event,
            timestamp=datetime.now()
        )
        
        # 5. 触发响应（如果需要）
        if threat_level in ['high', 'critical']:
            self._trigger_response(assessment)
        
        return assessment
    
    def _apply_detection_rules(self, event: dict) -> list:
        """应用检测规则"""
        detections = []
        
        for rule in self.detection_rules:
            if self._rule_matches(rule, event):
                detection = Detection(
                    rule_id=rule.id,
                    rule_name=rule.name,
                    severity=rule.severity,
                    description=rule.description,
                    confidence=rule.confidence
                )
                detections.append(detection)
        
        return detections
    
    def _detect_anomalies(self, event: dict) -> list:
        """检测异常行为"""
        anomalies = []
        
        for model in self.anomaly_models:
            score = model.predict(event)
            if score > model.threshold:
                anomaly = Detection(
                    rule_id=f"anomaly_{model.id}",
                    rule_name=f"异常检测: {model.name}",
                    severity='medium',
                    description=f"检测到异常行为，异常分数: {score:.2f}",
                    confidence=min(score, 1.0)
                )
                anomalies.append(anomaly)
        
        return anomalies
```

```python
# 自动化合规检查器
class ComplianceChecker:
    def __init__(self, compliance_framework):
        self.framework = compliance_framework
        self.check_results = {}
        
    def run_checks(self) -> ComplianceReport:
        """运行所有合规检查"""
        report = ComplianceReport(
            framework=self.framework.name,
            check_time=datetime.now(),
            overall_status='compliant'
        )
        
        for control in self.framework.controls:
            check_result = self._check_control(control)
            report.controls.append(check_result)
            
            if check_result.status == 'non_compliant':
                report.overall_status = 'non_compliant'
            elif check_result.status == 'partially_compliant' and report.overall_status == 'compliant':
                report.overall_status = 'partially_compliant'
        
        # 生成执行摘要
        report.summary = self._generate_summary(report)
        
        # 发送报告
        self._send_report(report)
        
        return report
    
    def _check_control(self, control: ComplianceControl) -> ControlCheckResult:
        """检查单个控制项"""
        check_result = ControlCheckResult(
            control_id=control.id,
            control_name=control.name,
            requirement=control.requirement
        )
        
        # 运行检查
        try:
            if control.check_type == 'automated':
                result = self._run_automated_check(control)
            elif control.check_type == 'manual':
                result = self._run_manual_check(control)
            elif control.check_type == 'evidence':
                result = self._check_evidence(control)
            else:
                result = CheckResult(
                    status='unknown',
                    details=f"未知检查类型: {control.check_type}"
                )
            
            check_result.status = result.status
            check_result.details = result.details
            check_result.evidence = result.evidence
            check_result.last_check = datetime.now()
            
        except Exception as e:
            check_result.status = 'error'
            check_result.details = f"检查执行失败: {str(e)}"
            check_result.last_check = datetime.now()
        
        return check_result
    
    def _run_automated_check(self, control: ComplianceControl) -> CheckResult:
        """运行自动化检查"""
        # 根据控制类型执行不同检查
        if control.category == 'access_control':

        elif control.category == 'encryption':
            return self._check_encryption(control)
        elif control.category == 'audit':
            return self._check_audit_logging(control)
```

```python
# 自动化响应剧本
class AutomatedResponsePlaybook:
    def __init__(self, playbooks):
        self.playbooks = playbooks
        
    def execute_playbook(self, incident: SecurityIncident) -> ResponseResult:
        """执行响应剧本"""
        # 1. 选择剧本
        playbook = self._select_playbook(incident)
        
        if not playbook:
            return ResponseResult(
                status='manual_required',
                message='无适用自动化剧本，需要人工响应'
            )
        
        # 2. 验证执行条件
        if not self._verify_execution_conditions(playbook, incident):
            return ResponseResult(
                status='conditions_not_met',
                message='执行条件未满足'
            )
        
        # 3. 执行响应步骤
        results = []
        for step in playbook.steps:
            try:
                step_result = self._execute_step(step, incident)
                results.append(step_result)
                
                if step_result.status == 'failed':
                    # 步骤失败，执行回滚
                    self._rollback_steps(results)
                    return ResponseResult(
                        status='failed',
                        message=f"步骤'{step.name}'执行失败",
                        details=step_result.details
                    )
                    
            except Exception as e:
                # 异常处理
                self._rollback_steps(results)
                return ResponseResult(
                    status='error',
                    message=f"步骤执行异常: {str(e)}"
                )
        
        # 4. 验证响应效果
        verification_result = self._verify_response_effectiveness(incident)
        
        return ResponseResult(
            status='completed',
            message='自动化响应剧本执行完成',
            details={
                'playbook': playbook.name,
                'steps_executed': len(results),
                'verification': verification_result
            }
        )
    
    def _execute_step(self, step: ResponseStep, incident: SecurityIncident) -> StepResult:
        """执行单个响应步骤"""
        if step.type == 'containment':
            return self._execute_containment_step(step, incident)
        elif step.type == 'eradication':
            return self._execute_eradication_step(step, incident)
        elif step.type == 'recovery':
            return self._execute_recovery_step(step, incident)
        elif step.type == 'notification':
            return self._execute_notification_step(step, incident)
        else:
            return StepResult(
                status='skipped',
                details=f"未知步骤类型: {step.type}"
            )
    
    def _execute_containment_step(self, step: ResponseStep, incident: SecurityIncident) -> StepResult:
        """执行遏制步骤"""
        if step.action == 'block_ip':
            # 阻止IP地址
            return self._block_ip_address(incident.source_ip, step.parameters)
        elif step.action == 'disable_account':
            # 禁用账户
            return self._disable_user_account(incident.user_id, step.parameters)
        elif step.action == 'isolate_host':
            # 隔离主机
            return self._isolate_host(incident.hostname, step.parameters)
        else:
            return StepResult(
                status='skipped',
                details=f"未知遏制动作: {step.action}"
            )
```

```python
class IntegrationSecurityTests:
    """集成层安全测试套件"""
    
    def test_authentication_bypass(self):
        """测试认证绕过漏洞"""
        # 尝试无认证访问
        response = self.client.get('/api/v1/integration/health')
        assert response.status_code == 401
        
        # 尝试无效令牌
        headers = {'Authorization': 'Bearer invalid_token'}
        response = self.client.get('/api/v1/integration/health', headers=headers)
        assert response.status_code == 401
        
        # 尝试过期令牌
        expired_token = self.create_expired_token()
        headers = {'Authorization': f'Bearer {expired_token}'}
        response = self.client.get('/api/v1/integration/health', headers=headers)
        assert response.status_code == 401
    
    def test_authorization_bypass(self):
        """测试授权绕过漏洞"""
        # 低权限用户尝试高权限操作
        low_priv_token = self.create_token(['integration:read'])
        
        headers = {'Authorization': f'Bearer {low_priv_token}'}
        
        # 尝试读取操作（应该成功）
        response = self.client.get('/api/v1/integration/config', headers=headers)
        assert response.status_code == 200
        
        # 尝试写入操作（应该失败）
        response = self.client.put('/api/v1/integration/config', 
                                  json={'some_config': 'value'},
                                  headers=headers)
        assert response.status_code == 403
    
    def test_injection_attacks(self):
        """测试注入攻击"""
        # SQL注入测试
        malicious_payload = {
            'filter': "1' OR '1'='1' --"
        }
        
        response = self.client.post('/api/v1/integration/events/search',
                                   json=malicious_payload)
        # 应该返回400或正常处理（参数化查询）
        assert response.status_code in [400, 200]
        if response.status_code == 200:
            # 验证没有异常数据返回
            assert 'error' not in response.json()
    
    def test_rate_limit_bypass(self):
        """测试速率限制绕过"""
        token = self.create_valid_token()

            response = self.client.get('/api/v1/integration/health', headers=headers)
            if response.status_code == 429:  # 太多请求

        # 应该有一定比例的请求被限制
```
